**src/main.ts** — Create the `src/` directory and then `src/main.ts` with these exact contents:
```typescript
import { Plugin } from 'obsidian';

export interface TogglImportSettings {
	apiToken: string;
	outputFormat: 'table' | 'plaintext';
	columns: {
		description: boolean;
		startTime: boolean;
		duration: boolean;
		tags: boolean;
		project: boolean;
	};
	delimiter: string;
}

export const DEFAULT_SETTINGS: TogglImportSettings = {
	apiToken: '',
	outputFormat: 'table',
	columns: {
		description: true,
		startTime: true,
		duration: true,
		tags: false,
		project: false,
	},
	delimiter: '|',
};

export default class TogglImportPlugin extends Plugin {
	settings!: TogglImportSettings;

	async onload(): Promise<void> {
		await this.loadSettings();
		// Phase 2: register settings tab
		// Phase 5: register import command
	}

	onunload(): void {}

	async loadSettings(): Promise<void> {
		this.settings = Object.assign(
			{},
			DEFAULT_SETTINGS,
			await this.loadData()
		);
	}

	async saveSettings(): Promise<void> {
		await this.saveData(this.settings);
	}
}
```

Critical requirements for src/main.ts:
- `apiToken` defaults to empty string `''` — prevents accidental API calls
- `Object.assign({}, DEFAULT_SETTINGS, await this.loadData())` — THREE arguments, empty object first. This prevents mutating DEFAULT_SETTINGS.
- `settings!: TogglImportSettings` uses definite assignment assertion (! suffix) because settings is assigned in loadSettings() called from onload()
- `onload()` calls ONLY `this.loadSettings()` — no addCommand, no addSettingTab, no Notice
- Plugin class is the default export
- TogglImportSettings interface and DEFAULT_SETTINGS are named exports (used by Phase 2 settings tab)
- Use tabs for indentation (Obsidian plugin convention)
- Do NOT import `requestUrl` in this file — it is not used until Phase 3 API client. The import will exist in a separate api module.
